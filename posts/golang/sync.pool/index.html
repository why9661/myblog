  6func poolCleanup() {
  7	// 将每个P的victime清空
  8	for _, p := range oldPools {
  9		p.victim = nil
 10		p.victimSize = 0
 11	}
 12
 13	// 将local中的对象移动到victim中，清空local
 14	for _, p := range allPools {
 15		p.victim = p.local
 16		p.victimSize = p.localSize
 17		p.local = nil
 18		p.localSize = 0
 19	}
 20
 21	oldPools, allPools = allPools, nil
 22}

 24type Pool struct {
 25	noCopy noCopy
 26	local     unsafe.Pointer 
 27	localSize uintptr       
 28	victim     unsafe.Pointer 
 29	victimSize uintptr        
 30	New func() interface{}
 31}
 32
 33//每个P维护一个本地的poolLocal
 34type poolLocalInternal struct {
 35    //poolLocal中维护一个private和一个shared队列
 36    //private只能给当前P上运行的goroutine使用
 37    //shared可以被其它P上运行的goroutine获取
 38	private interface{} 
 39	shared  poolChain   
 40}
 41
 42type poolLocal struct {
 43	poolLocalInternal
 44	pad [128 - unsafe.Sizeof(poolLocalInternal{})%128]byte
 45}

 47// Get流程
 48func (p *Pool) Get() interface{} {
 49	...
 50    // 将goroutine绑定到当前P上，不能被抢占
 51	l, pid := p.pin()
 52	x := l.private
 53	l.private = nil
 54    //1、尝试从本地P的pool的private获取
 55	if x == nil {
 56        //2、如果本地P的pool的private为空，那么尝试从shared获取（从队列头部获取）
 57		x, _ = l.shared.popHead()
 58        //3、如果本地的P的pool的shared队列也为空，那么进入getSlow流程
 59		if x == nil {
 60			x = p.getSlow(pid)
 61		}
 62	}
 63    // 解除抢占
 64	runtime_procUnpin()
 65    
 66	...
 67    // 6、如果都没获取到，那么执行传入的New方法返回一个新的对象
 68	if x == nil && p.New != nil {
 69		x = p.New()
 70	}
 71	return x
 72}
 73
 74func (p *Pool) getSlow(pid int) interface{} {
 75	size := runtime_LoadAcquintptr(&p.localSize) 
 76	locals := p.local                            
 77	// 4、尝试从其它P的pool的shared队列获取（从队列尾部获取）
 78	for i := 0; i < int(size); i++ {
 79		l := indexLocal(locals, (pid+i+1)%int(size))
 80		if x, _ := l.shared.popTail(); x != nil {
 81			return x
 82		}
 83	}
 84
 85	// 5、尝试从victim获取
 86	size = atomic.LoadUintptr(&p.victimSize)
 87	if uintptr(pid) >= size {
 88		return nil
 89	}
 90	locals = p.victim
 91	l := indexLocal(locals, pid)
 92	if x := l.private; x != nil {
 93		l.private = nil
 94		return x
 95	}
 96	for i := 0; i < int(size); i++ {
 97		l := indexLocal(locals, (pid+i)%int(size))
 98		if x, _ := l.shared.popTail(); x != nil {
 99			return x
100		}
101	}
102
103	atomic.StoreUintptr(&p.victimSize, 0)
104
105	return nil
106}

108// Put流程
109func (p *Pool) Put(x interface{}) {
110	if x == nil {
111		return
112	}
113	...
114    // 将goroutine绑定到当前P上，不能被抢占
115	l, _ := p.pin()
116    // 1、如果p的poolLocal的private为空，那么就存到private
117	if l.private == nil {
118		l.private = x
119		x = nil
120	}
121    // 2、否则存到shared队列头部
122	if x != nil {
123		l.shared.pushHead(x)
124	}
125    // 解除抢占
126	runtime_procUnpin()
127	...
128}
